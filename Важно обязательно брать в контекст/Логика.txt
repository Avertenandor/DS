def find_block_by_timestamp(target_ts: int) -> int:
    left, right = 1, w3.eth.block_number
    for _ in range(50):
        mid = (left + right) // 2
        blk = w3.eth.get_block(mid)
        if blk.timestamp < target_ts:
            left = mid + 1
        else:
            right = mid - 1
    return max(1, right)
```

### 2.4. –°–±–æ—Ä –ª–æ–≥–æ–≤ —á–∞–Ω–∫–∞–º–∏ —Å –∞–¥–∞–ø—Ç–∏–≤–Ω—ã–º —Ä–∞–∑–º–µ—Ä–æ–º –∏ —Ä–µ—Ç—Ä–∞—è–º–∏

import time

def get_transfer_logs(token_addr: str, start: int, stop: int):
    all_logs = []
    chunk = 1000
    block = start

        end = min(block + chunk - 1, stop)
        params = {
            "fromBlock": hex(block),
            "toBlock":   hex(end),
            "address":   Web3.toChecksumAddress(token_addr),
            "topics":    [event_sig, None, topic_to]
        }

        # –î–æ 3 –ø–æ–ø—ã—Ç–æ–∫ –Ω–∞ –æ—à–∏–±–∫—É

            try:
                logs = w3.eth.get_logs(params)
                break
            except Exception as e:
                err = str(e).lower()
                backoff = 2 ** (attempt - 1)
                time.sleep(backoff)
        else:
            raise RuntimeError(f"–ù–µ —É–¥–∞–ª–æ—Å—å –ø–æ–ª—É—á–∏—Ç—å –ª–æ–≥–∏ {block}-{end}: {err}")

        # –£—Å–ø–µ—à–Ω–æ –ø–æ–ª—É—á–∏–ª–∏ ‚Äî –∑–∞–¥–µ—Ä–∂–∫–∞ –ø–µ—Ä–µ–¥ —Å–ª–µ–¥—É—é—â–∏–º –∑–∞–ø—Ä–æ—Å–æ–º
        time.sleep(REQUEST_DELAY)

        # –ê–¥–∞–ø—Ç–∏–≤–Ω–∞—è –ª–æ–≥–∏–∫–∞ —Ä–∞–∑–º–µ—Ä–∞ —á–∞–Ω–∫–∞
        if not logs and chunk < MAX_CHUNK:
            chunk = min(chunk * 2, MAX_CHUNK)
        elif len(logs) > 2000 and chunk > MIN_CHUNK:
            chunk = max(chunk // 2, MIN_CHUNK)

        # –û–±—Ä–∞–±–æ—Ç–∫–∞
        for log in logs:
            frm = Web3.toChecksumAddress("0x" + log["topics"][1].hex()[-40:])
            val = int(log["data"], 16) / 10**9  # DECIMALS = 9
            all_logs.append({
                "block": log["blockNumber"],
                "from":  frm,
                "value": val,
                "tx":    log["transactionHash"].hex()
            })

        block = end + 1

    return all_logs
```
